# End of the Tahitian Civil War (Nationalist victory)
news_event = {
	id = TAH.news.64
	title = TAH.news.64.t
	desc = TAH.news.64.d
	picture = GFX_news_event_033
	
	major = yes
	
	trigger = {
		has_global_flag = tahitian_civil_war
		NOT = { has_global_flag = tcw_over }
		TAH = { has_civil_war = no }
		any_other_country = {
			original_tag = TAH
			has_government = fascism
		}
		all_country = {
			OR = {
				NOT = { original_tag = TAH }
				has_government = fascism
			}
		}
	}
	
	mean_time_to_happen = {
		days = 2
	}
	
	immediate = {
		set_global_flag = tah_nationalist_victory
		set_global_flag = tcw_over
	}
	
	option = {
		name = TAH.news.64.a
		trigger = {
			NOT = { original_tag = TAH }
		}
	}
	option = {
		name = TAH.news.64.b
		trigger = { original_tag = TAH }
		delete_unit_template_and_units = { division_template = "Brigada Legionario" }
	}
}

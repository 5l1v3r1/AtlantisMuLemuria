		tag = FSM

			AND = {
				OR = {
							controls_state = 924

							controls_state = 923
							}
				TIQ = {
					has_war_with = FSM
					has_capitulated = yes
					is_in_faction = yes
					}
				}
			AND = {
				OR = {
							controls_state = 924
							controls_state = 925
							controls_state = 926
							controls_state = 923
							}
				NOT = {
					country_exists = TIQ
					}
				}
			}
		}

	option = {
		ai_chance = { factor = 100 }
		name = fsm.11.a
		puppet = TIQ
		if = {
		limit = {
			 controls_state = 923
		}
		transfer_state = 923
		923 = { set_state_name = "Jillilstaat" }
		}



		TIQ ={
	if = {
		limit = {
			FSM = { controls_state = 924 } 
		}
		transfer_state = 924
		}
	if = {
		limit = {
			FSM = { controls_state = 925 } 
		}
		transfer_state = 925
		}
	if = {
		limit = {
			FSM = { controls_state = 926 } 
		}
		transfer_state = 926
		}

		}

	
	}
	option = {
		ai_chance = { factor = 100 }
		name = fsm.11.b
		if = {
		limit = {
			 controls_state = 923
		}
		transfer_state = 923
		923 = { set_state_name = "Jillilstaat" }
		}

	}

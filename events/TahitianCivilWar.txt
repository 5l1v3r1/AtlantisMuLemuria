add_namespace = tahiti

# Tahitian Civil War
country_event = {
	id = tahiti.1
	title = tahiti.1.t
	desc = tahiti.1.d
	picture = GFX_report_event_spain_civil_war_soldiers
	
	fire_only_once = yes
	
	trigger = {
		NOT = { has_global_flag = tcw_over }
		tag = TAH
		has_defensive_war = no
	}

	mean_time_to_happen = {
		days = 60
	}
	
	option = { # Support Nationalists
		name = tahiti.1.a
		trigger = { is_ai = no }
		hidden_effect = {
			start_civil_war = {
				ruling_party = democratic
				ideology = communism
				size = 0.5
				capital = 641
				states = { 641 897 896 }
			}
			random_other_country = {
				limit = {
					original_tag = TAH
					has_government = communism
				}
				load_oob = "TAH_Civil_War"
				set_equipment_fraction = 0.5
				set_stability = 0.9
				set_war_support = 0.8
				country_event = { id = tahiti.3 days = 1 }
				add_popularity = {
					ideology = communism
					popularity = 1
				}
			}
			news_event = { hours = 6 id = tah.news.63 }
			set_global_flag = tahitian_civil_war
			
			load_oob = "TTI_civil_war"
			set_equipment_fraction = 0.5
			set_stability = 0.9
			set_war_support = 0.8
			
			add_popularity = {
				ideology = democratic
				popularity = 1
			}
			random_other_country = {
				limit = {
					original_tag = TAH
					has_government = communism
				}
				set_cosmetic_tag = TAH_civil_war
			}

			country_event = { id = tahiti.2 days = 7 }
		}
		custom_effect_tooltip = tcw_nationalists_chosen
	}
	option = { # Support Communists
		name = tahiti.1.b
		hidden_effect = {
			set_capital = 899
			start_civil_war = {
				ruling_party = communism
				ideology = democratic
				size = 0.5
				capital = 641
				states = { 641 897 896 }
			}
			random_other_country = {
				limit = {
					original_tag = TAH
					has_government = democratic
				}
				set_equipment_fraction = 0.5
				set_stability = 0.9
				set_war_support = 0.8
				country_event = { id = tahiti.2 days = 1 }
				load_oob = "TTI_civil_war"
			}
			news_event = { hours = 6 id = tah.news.63 }
			set_global_flag = tahitian_civil_war
			
			set_politics = {
				ruling_party = communism
				elections_allowed = no
			}
			add_popularity = {
				ideology = communism
				popularity = 0.5
			}
			add_popularity = {
				ideology = democratic
				popularity = -1
			}	
			random_other_country = {
				limit = {
					original_tag = TAH
					has_government = democratic
				}
				set_cosmetic_tag = TTI_civil_war
			}
			load_oob = "TAH_civil_war"
			set_equipment_fraction = 0.5
			set_stability = 0.9
			set_war_support = 0.8
			
			country_event = { id = tahiti.3 days = 7 }
		}
		custom_effect_tooltip = tcw_republicans_chosen
	}
}

# Election of 1936
country_event = {
	id = tahiti.9
	title = tahiti.9.t
	desc = tahiti.9.d
	picture = GFX_report_event_election_vote
	
	is_triggered_only = yes

	fire_only_once = yes

	trigger = {
		tag = TAH
		date < 1937.1.1
	}

	# Popular Front
	option = {
		ai_chance = { factor = 1 }
		name = tahiti.9.a
		add_popularity = {
			ideology = communism
			popularity = 0.1
		}
		set_politics = {
			ruling_party = communism
			elections_allowed = no
		}
	}

	# Falange
	option = {
		ai_chance = { factor = 0 }
		name = tahiti.9.b
		set_politics = {
			ruling_party = democratic
			elections_allowed = no
		}
		set_country_flag = TAH_communist_civil_war
	}
}
